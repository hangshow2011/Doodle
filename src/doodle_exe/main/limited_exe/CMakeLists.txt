add_executable(
        doodle_limited_exe
        WIN32
        main.cpp
        gui/app.cpp
        gui/app.h
        gui/window.h
        gui/window.cpp
        ../DoodleExe.manifest
        ../icon.ico
        ../win_exe.rc
)

        PROPERTIES RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

        PUBLIC
        doodle_lib
        )

        PUBLIC
        #            _WIN32_WINNT=0x0A00

        ISOLATION_AWARE_ENABLED
        $<$<CONFIG:Debug>:SPDLOG_ACTIVE_LEVEL=1>
